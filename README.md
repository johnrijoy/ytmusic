# ytmusic
Go library for searching on YouTube Music. Work in progress.

```go
package main

import (
	"encoding/json"
	"fmt"
	"github.com/raitonoberu/ytmusic"
)

func main() {
	s := ytmusic.Search("summer vibes")
	result, err := s.Next()
	if err != nil {
		panic(err)
	}

	jsonstr, _ := json.Marshal(result)
	fmt.Println(string(jsonstr))
}
```
